#include <stdio.h>

int get_intersection(int x1, int y1, int r1, int x2, int y2, int r2);
int main()
{
	int how_many;
	int x1 = 0, x2 = 0, y1 = 0, y2 = 0, r1 = 0, r2 = 0;
	int judge;

	scanf_s("%d", &how_many);
	while (how_many-- > 0) {
		scanf_s("%d %d %d %d %d %d", &x1, &y1, &r1, &x2, &y2, &r2);
		judge = get_intersection(x1, y1, r1, x2, y2, r2);
		if (judge > 2)
			printf("-1\n");
		else
			printf("%d\n", judge);
	}
	return 0;
}

int get_intersection(int x1, int y1, int r1, int x2, int y2, int r2)
{
	int xi, yi, X, Y;
	int count = 0;

	for (xi = (x1 - r1); xi <= (x1 + r1); xi++)
	for (yi = (y1 - r1); yi <= (y1 + r1); yi++)
	if (((xi - x1)*(xi - x1) + (yi - y1)*(yi - y1)) == (r1*r1)) {
		X = xi;
		Y = yi;
		if (((X - x2)*(X - x2) + (Y - y2)*(Y - y2)) == (r2*r2))
			count++;
	}

	return count;
}
